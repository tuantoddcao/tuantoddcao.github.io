  freq <- table(x)
  percentage <- prop.table(freq)*100
  percentage <- round(percentage, 2)
  result <- data.frame(
    value = names(freq),
    Frequency = as.vector(freq),
    Percentage = as.vector(percentage)
  )
  return(result)
}

make_ml_data <- function(data_noout) {
  df <- data.frame(
    E.use = rowMeans(data_noout[, c("easyuse_1","easyuse_2","easyuse_3")], na.rm = TRUE),
    Personalization = rowMeans(data_noout[, c("personal_1","personal_2","personal_3","personal_4")], na.rm = TRUE),
    R.advantage = rowMeans(data_noout[, c("advantage_1","advantage_2","advantage_3")], na.rm = TRUE),
    S.visibility = rowMeans(data_noout[, c("visibility_1","visibility_2","visibility_3")], na.rm = TRUE),
    P.compatibility = rowMeans(data_noout[, c("compatibility_1","compatibility_2","compatibility_3","compatibility_4")], na.rm = TRUE),
    Reliance = rowMeans(data_noout[, c("reliance_1","reliance_2","reliance_3")], na.rm = TRUE),
    Purchase.intent = rowMeans(data_noout[, c("purchase_1","purchase_2","purchase_3")], na.rm = TRUE),
    Ad.skepticism = rowMeans(data_noout[, c("skepticism_1","skepticism_2","skepticism_3","skepticism_4","skepticism_5")], na.rm = TRUE)
  )

  df <- na.omit(df)

  # mean-center for interaction (no scaling)
  df[, c("Reliance","Ad.skepticism")] <- scale(df[, c("Reliance","Ad.skepticism")],
                                              center = TRUE, scale = FALSE)
  df$Reliance_x_Ad.skepticism <- df$Reliance * df$Ad.skepticism
  df
}

## 1) Elastic Net (10-fold CV) + final refit at lambda.1se

fit_enet <- function(X, y, alpha = 0.5, nfolds = 10, seed = 123) {
  set.seed(seed)

  cv_fit <- cv.glmnet(
    x = X, y = y,
    alpha = alpha,
    nfolds = nfolds,
    standardize = TRUE,
    keep = TRUE
  )
  idx <- which(cv_fit$lambda == cv_fit$lambda.1se)
  pred_oof <- as.numeric(cv_fit$fit.preval[, idx])

  rmse <- sqrt(mean((y - pred_oof)^2, na.rm = TRUE))
  mae  <- mean(abs(y - pred_oof), na.rm = TRUE)
  r2   <- cor(y, pred_oof, use = "complete.obs")^2

  final_fit <- glmnet(
    x = X, y = y,
    alpha = alpha,
    lambda = cv_fit$lambda.1se,
    standardize = TRUE
  )

  list(
    cv = cv_fit,
    lambda_1se = cv_fit$lambda.1se,
    oof = list(rmse = rmse, mae = mae, r2 = r2, pred = pred_oof),
    final = final_fit,
    coef = coef(final_fit)
  )
}

## Random Forest (caret 10-fold CV) + variable importance

fit_rf_cv <- function(X, y, k = 10, tuneLength = 5, ntree = 1000, seed = 123) {
  set.seed(seed)

  ctrl <- trainControl(method = "cv", number = k)

  rf_cv <- train(
    x = X, y = y,
    method = "rf",
    trControl = ctrl,
    tuneLength = tuneLength,
    ntree = ntree,
    importance = TRUE
  )

  list(
    cv = rf_cv,
    results = rf_cv$results[, c("mtry","RMSE","Rsquared","MAE")],
    best = rf_cv$bestTune,
    final_model = rf_cv$finalModel,
    varimp = varImp(rf_cv, scale = TRUE)
  )
}

rf_res <- fit_rf_cv(X, y, k = 10, tuneLength = 5, ntree = 1000, seed = 123)

rf_perm_importance_cv <- function(X, y, seed = 123, k = 10, ntree = 1000, tuneLength = 5) {
  set.seed(seed)

  folds <- createFolds(y, k = k, returnTrain = TRUE)
  ctrl <- trainControl(method = "cv", number = k, index = folds)

  # tune once to get best mtry under the same CV setup
  rf_cv <- train(
    x = X, y = y,
    method = "rf",
    trControl = ctrl,
    tuneLength = tuneLength,
    ntree = ntree,
    importance = TRUE
  )
  best_mtry <- rf_cv$bestTune$mtry

  inc_rmse <- sapply(colnames(X), function(v) {
    fold_increases <- numeric(length(folds))

    for (i in seq_along(folds)) {
      tr_idx <- folds[[i]]
      te_idx <- setdiff(seq_len(nrow(X)), tr_idx)

      Xtr <- X[tr_idx, , drop = FALSE]
      ytr <- y[tr_idx]
      Xte <- X[te_idx, , drop = FALSE]
      yte <- y[te_idx]

      set.seed(seed + i)
      rf_fit <- randomForest(x = Xtr, y = ytr, ntree = ntree, mtry = best_mtry)

      pred_base <- predict(rf_fit, Xte)
      rmse_base <- sqrt(mean((yte - pred_base)^2))

      Xte_perm <- Xte
      Xte_perm[, v] <- sample(Xte_perm[, v])
      pred_perm <- predict(rf_fit, Xte_perm)
      rmse_perm <- sqrt(mean((yte - pred_perm)^2))

      fold_increases[i] <- rmse_perm - rmse_base
    }

    mean(fold_increases)
  })

  permu <- data.frame(Predictor = names(inc_rmse), IncRMSE = as.numeric(inc_rmse))
  permu <- permu[order(-permu$IncRMSE), ]

  list(best_mtry = best_mtry, permutation_importance = permu, rf_cv = rf_cv)
}
